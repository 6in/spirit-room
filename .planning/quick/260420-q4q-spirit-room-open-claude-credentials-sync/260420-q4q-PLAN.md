```bash
# ── host 認証情報をボリュームへ同期 (open/kaio 共用) ─────────
# Claude Code の OAuth トークンは短命なので、部屋起動ごとに host の
# 最新 ~/.claude/.credentials.json を AUTH_VOLUME の .credentials.json に
# 上書きコピーする。host に creds が無い場合は WARN のみで return し、
# 呼び出し側 (open/kaio) はそのまま docker run -d へ進める。
_sync_host_credentials() {
    local host_creds="${HOME}/.claude/.credentials.json"
    if [ ! -f "$host_creds" ]; then
        echo "[WARN] ホストに認証情報なし: $host_creds"
        echo "       spirit-room auth を先に実行してください"
        return 0
    fi

    echo "[INFO] ホストの最新認証情報をボリュームに同期中..."
    docker run --rm \
        -v "${AUTH_VOLUME}:/dst" \
        -v "${host_creds}:/src/.credentials.json:ro" \
        -e HOST_UID="$(id -u)" \
        -e HOST_GID="$(id -g)" \
        --entrypoint /bin/bash \
        "$BASE_IMAGE" \
        -c "cp /src/.credentials.json /dst/.credentials.json && chown \$HOST_UID:\$HOST_GID /dst/.credentials.json && chmod 600 /dst/.credentials.json" \
        > /dev/null 2>&1 || echo "[WARN] 認証情報同期に失敗"
}
```
