**変更 2: log() 関数の直後 (現行 line 30 の後) に max_iterations 解決関数を追加**

```bash
log() { echo "[$(date '+%Y-%m-%d %H:%M:%S')] $*" | tee -a "$LOG_FILE"; }

# ── MAX_ITERATIONS 解決 (D-11 優先順: env > MISSION.md frontmatter > default) ─
# 出力: stdout に整数 1 行
resolve_max_iterations() {
    # ① env MAX_ITERATIONS が整数なら最優先
    if [ -n "${MAX_ITERATIONS:-}" ] && [[ "${MAX_ITERATIONS}" =~ ^[0-9]+$ ]]; then
        echo "$MAX_ITERATIONS"
        return
    fi

    # ② MISSION.md 先頭 frontmatter から拾う (yq を使わず sed+grep で堅牢に)
    if [ -f "$MISSION_FILE" ]; then
        # 先頭が --- で始まる frontmatter ブロックを抽出し、max_iterations: N 行を拾う
        local _fm_val
        _fm_val=$(sed -n '/^---$/,/^---$/p' "$MISSION_FILE" 2>/dev/null \
            | grep -E '^max_iterations:[[:space:]]*[0-9]+' \
            | head -1 \
            | awk -F: '{gsub(/[[:space:]]/, "", $2); print $2}')
        if [ -n "$_fm_val" ] && [[ "$_fm_val" =~ ^[0-9]+$ ]]; then
            echo "$_fm_val"
            return
        fi
    fi

    # ③ デフォルト
    echo "$DEFAULT_MAX_ITERATIONS"
}
```
